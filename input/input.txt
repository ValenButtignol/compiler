Program {
    def integer inc(integer x) {
        x = x + 1;
        return x + 1;
    }

    def void main() {
        integer y = 0;
        integer a = 1;
        a = inc(a);
        if (-a) then {
            y = 2;
        } else {
            while (y > 5) {
                y = y + 1;
            }
        }
    }
}
